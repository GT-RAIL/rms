 43 /**
 44  * Creates a modal overlay with a loading icon. Disable the screen with removeModalPageLoading().
 45  */
 46 function createModalPageLoading() {
 47   // remove any old modals that may exist
 48   if ($('#LOADING-DIV')) {
 49     $('#LOADING-DIV').remove();
 50   }
 51 
 52   $('body').append('<div id="LOADING-DIV" class="modal"></div>');
 53   $('body').addClass('loading');
 54 }
 55 
 56 /**
 57  * Disable the loading modal overlay.
 58  */
 59 function removeModalPageLoading() {
 60   $('#LOADING-DIV').remove();
 61   $('body').removeClass('loading');
 62 }

 64 /**
 65  * Creates a displays a modal dialog with the given error message.
 66  * 
 67  * @param {string}
 68  *            message the error message to display
 69  */
 70 function createErrorDialog(message) {
 71   // remove any old dialogs that may exist
 72   if ($('#ERROR-DIALOG')) {
 73     $('#ERROR-DIALOG').remove();
 74   }
 75 
 76   // create an error div
 77   var html = '<div id="ERROR-DIALOG"><b>';
 78   html += '<span class="ui-icon ui-icon-alert" style="float: left; margin: 0 7px 50px 0;"></span>';
 79   html += message;
 80   html += '</b></div>';
 81   $('body').append(html);
 82 
 83   $('#ERROR-DIALOG').dialog({
 84     title : 'Error!',
 85     draggable : false,
 86     resizable : false,
 87     dialogClass : 'alert',
 88     modal : true,
 89     show : 'fade',
 90     hide : 'fade',
 91     buttons : {
 92       Ok : function() {
 93         $(this).dialog('close');
 94       }
 95     }
 96   });
 97 }

130 /**
131  * A function to check if the given rosbridge server is online
132  * 
133  * @param {String}
134  *            protocol the protocol to use
135  * @param {String}
136  *            host the hostname of the rosbridge server
137  * @param {int}
138  *            port the port of the rosbridge server
139  * @param {function}
140  *            callback the callback function returning if the server is online
141  */
142 function rosonline(protocol, host, port, callback) {
143   var ros = new ROS(protocol + host + ':' + port);
144   ros.on('connection', function() {
145     callback(true);
146   });
147   ros.on('error', function() {
148     callback(false);
149   });
150 }

152 /**
153  * Makes a logout AJAX request.
154  */
155 function logout() {
156   createModalPageLoading();
157 
158   // create a AJAX request
159   var formData = new FormData();
160   formData.append('request', 'destroy_session');
161   $.ajax('../../api/users/user_accounts/', {
162     data : formData,
163     cache : false,
164     contentType : false,
165     processData : false,
166     type : 'POST',
167     beforeSend : function(xhr) {
168       // authenticate with the header
169       xhr.setRequestHeader('RMS-Use-Session', 'true');
170     },
171     success : function(data) {
172       // success
173       window.location = '../';
174     }
175   });
176 }

178 /**
179  * Base 64 encode the given string.
180  * 
181  * @param {String}
182  *            string the string to base 64 encode
183  * @returns {String} the base 64 encoded string
184  */
185 function base64Encode(string) {
186   var key = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=';
187 
188   // convert to utf8
189   string = string.replace(/\r\n/g, '\n');
190   var utf = '';
191   for ( var n = 0; n < string.length; n++) {
192     var c = string.charCodeAt(n);
193     if (c < 128) {
194       utf += String.fromCharCode(c);
195     } else if ((c > 127) && (c < 2048)) {
196       utf += String.fromCharCode((c >> 6) | 192);
197       utf += String.fromCharCode((c & 63) | 128);
198     } else {
199       utf += String.fromCharCode((c >> 12) | 224);
200       utf += String.fromCharCode(((c >> 6) & 63) | 128);
201       utf += String.fromCharCode((c & 63) | 128);
202     }
203   }
204 
205   // create the output buffer
206   var output = '';
207   var chr1, chr2, chr3, enc1, enc2, enc3, enc4;
208   var i = 0;
209   // parse and convert the string
210   while (i < utf.length) {
211     chr1 = utf.charCodeAt(i++);
212     chr2 = utf.charCodeAt(i++);
213     chr3 = utf.charCodeAt(i++);
214     enc1 = chr1 >> 2;
215     enc2 = ((chr1 & 3) << 4) | (chr2 >> 4);
216     enc3 = ((chr2 & 15) << 2) | (chr3 >> 6);
217     enc4 = chr3 & 63;
218 
219     if (isNaN(chr2)) {
220       enc3 = enc4 = 64;
221     } else if (isNaN(chr3)) {
222       enc4 = 64;
223     }
224 
225     // add to the output
226     output = output + key.charAt(enc1) + key.charAt(enc2) + key.charAt(enc3)
227         + key.charAt(enc4);
228   }
229 
230   return output;
231 }
