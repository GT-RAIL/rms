 12 /**
 13  * Log the given message into the study_logs MySQL table. This function will only work if the _EXPID
 14  * global variable is defined.
 15  * 
 16  * @param mesasge
 17  *            {string} the message to log
 18  */
 19 function studyLog(message) {
 20   if (typeof (_EXPID) !== 'undefined') {
 21     // create an AJAX request
 22     var data = new FormData();
 23     data.append('expid', _EXPID);
 24     data.append('entry', message);
 25     $.ajax('../../api/user_studies/study_logs/', {
 26       data : data,
 27       cache : false,
 28       contentType : false,
 29       processData : false,
 30       type : 'POST',
 31       beforeSend: function (xhr) {
 32         // authenticate with the header
 33         xhr.setRequestHeader('RMS-Use-Session', 'true');
 34       }
 35     });
 36   }
 37 }
